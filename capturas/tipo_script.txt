CREATE DATABASE powertechsolutions;

USE powertechsolutions;

CREATE TABLE IF NOT EXISTS processos (
id_proc INT AUTO_INCREMENT PRIMARY KEY,
nome VARCHAR(80),
tempo_user DOUBLE
);

CREATE TABLE IF NOT EXISTS analise_desempenho(
id INT AUTO_INCREMENT PRIMARY KEY,
titulo_janela VARCHAR(100),
fkprocesso INT,
FOREIGN KEY (fkprocesso) REFERENCES processos (id_proc)
);

SELECT id_proc FROM processos;

SELECT * FROM analise_desempenho JOIN processos ON fkprocesso=id_proc
